Main Code



 


	for(samples = MIN_AVAIL(); samples>0; --samples)
	{
		IT_IN(0);
		IT_IN(1);
		in_d = d(0);
		if(cp(0) > 0.5 && cp(1) == 0.0)
		{
			if(in_d == 0.0)
			{
				temp_q = 0.0;
				temp_qp = 1.0;
			}

			if(in_d > 0.5)
			{
				temp_q = 1.0;
				temp_qp = 0.0;
			}
		}

		if(IT_OUT(0)) {
			KrnOverflow("dff",0);
			return(99);
		}
		q(0) = temp_q;
		if(IT_OUT(1)) {
			KrnOverflow("dff",1);
			return(99);
		}
		qp(0) = temp_qp;
	}
	return(0);
