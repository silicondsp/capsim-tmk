Main Code



 


if (numberOfRows >= matrixDim) return(0);

for(i=0; i< matrixDim; i++) {

	if(IT_OUT(0)) {
			KrnOverflow("xygen",0);
			return(99);
	}
	ramp = i*xStep+xMin;
	x(0) = ramp; 
	if(IT_OUT(1)) {
			KrnOverflow("xygen",1);
			return(99);
	}
	step= numberOfRows*yStep + yMin;
	y(0)=step;
}
numberOfRows++;
	
         
return(0);
