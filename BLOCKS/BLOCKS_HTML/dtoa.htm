Main Code



 

 
	/* note the minimum number of samples on the input 	*/
	/* buffers and iterate that many times 			*/
 
	for(no_samples=MIN_AVAIL();no_samples >0; --no_samples) 
 
		{
		if(IT_OUT(0)) {
			KrnOverflow("dtoa",0);
			return(99);	
		}
		IT_IN(0);	
		sample = x(0);
	 	sample = sample/(float)levels*maxLevel;	
		if (sample > maxLevel ) sample = maxLevel; 
		if (sample < -maxLevel ) sample = -maxLevel; 

		   y(0) = sample;  
		}
	return(0);
